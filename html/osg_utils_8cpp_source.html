00041 std::vector<osg::Node*> findNamedNodes(const std::string& searchName,
00042                                        osg::Node* currNode)
00043 {
00044     osg::Group* currGroup;
00045     std::vector<osg::Node*> foundNodes;
00046     osg::Node *foundNode;
00047 
00048     // check to see if we have a valid (non-NULL) node.
00049     // if we do have a null node, return NULL.
00050     if ( !currNode)
00051     {
00052         return foundNodes;
00053     }
00054 
00055     // We have a valid node, check to see if this is the node we
00056     // are looking for. If so, return the current node.
00057     if (currNode->getName() == searchName)
00058     {
00059         foundNodes.push_back(currNode);
00060     }
00061 
00062     // We have a valid node, but not the one we are looking for.
00063     // Check to see if it has children (non-leaf node). If the node
00064     // has children, check each of the child nodes by recursive call.
00065     // If one of the recursive calls returns a non-null value we have
00066     // found the correct node, so return this node.
00067     // If we check all of the children and have not found the node,
00068     // return NULL
00069     currGroup = currNode->asGroup(); // returns NULL if not a group.
00070     if ( currGroup )
00071     {
00072         for (unsigned int i = 0 ; i < currGroup->getNumChildren(); i ++)
00073         {
00074             std::vector<osg::Node*> subNodes = findNamedNodes(searchName, currGroup->getChild(i));
00075             if (subNodes.size())
00076                 for(int j=0; j<subNodes.size(); j++) foundNodes.push_back(subNodes[j]); // found a match!
00077         }
00078     }
00079     return foundNodes;
00080 }

00302 void Ellipsoid::draw()
00303 {
00304     myNormals->clear();
00305     myVertices->clear();
00306     myPrimitive->clear();
00307 
00308     osg::Vec3 nadir = myCenter+osg::Vec3(0,0,myRadii.z());
00309     osg::Vec3 zenith = myCenter+osg::Vec3(0,0,-myRadii.z());
00310 
00311     double hStep = 2*M_PI/myHBands, vStep = M_PI/myVBands;
00312     for (double theta=0; theta<M_PI; theta+=hStep)
00313     {
00314         for (double phi=0; phi<2*M_PI; phi+=vStep)
00315         {
00316             if (phi != 0 && phi!= M_PI)
00317             {
00318                 addPoint(osg::Vec3(myRadii.x()*sin(phi)*cos(theta)+myCenter.x(),
00319                                    myRadii.y()*sin(phi)*sin(theta)+myCenter.y(),
00320                                    myRadii.z()*cos(phi)+myCenter.z()));
00321                 addPoint(osg::Vec3(myRadii.x()*sin(phi)*cos(theta+hStep)+myCenter.x(),
00322                                    myRadii.y()*sin(phi)*sin(theta+hStep)+myCenter.y(),
00323                                    myRadii.z()*cos(phi)+myCenter.z()));
00324             }
00325             else if (phi == 0)
00326             {
00327                 addPoint(nadir);
00328                 addPoint(nadir);
00329             }
00330             else if (phi == M_PI)
00331             {
00332                 addPoint(zenith);
00333                 addPoint(zenith);
00334             }
00335             else std::cout<<"Error: Ellipsoid point not added"<<std::endl;
00336         }
